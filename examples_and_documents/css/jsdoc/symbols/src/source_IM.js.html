  7 	 */
  8 	var bDebug = false,
  9 		/**
 10 		 * bAsynchronous is a private flag to know if we want to execute the comparison using asynchronous mode or not.
 11 		 * @private
 12 		 * @type Boolean
 13 		 */
 14 		bAsynchronous = false,
 15 		/**
 16 		 * Number of image that are loaded not important to know if are loaded or with error.
 17 		 * @private
 18 		 * @type Number
 19 		 */
 20 		nImagesLoaded = 0,

 22 		 * Array of canvas that are created dynamically.
 23 		 * @private
 24 		 * @type Array
 25 		 */
 26 		aCanvas = [],
 27 		/**
 28 		 * fpLoop is a method that will save the asynchronous or not loop type

 30 		 * @type Function

 33 	/**
 34 	 * loopWithoutBlocking is a function to process items in asynchronous mode to avoid the environment to be freeze.
 35 	 * @private
 36 	 * @param aItems {Array} Items array to traverse.
 37 	 * @param fpProcess {Function} Callback to execute on each iteration.
 38 	 * @param fpFinish {Function} Callback to execute when all the items are traversed.
 39 	 */
 40 	function loopWithoutBlocking(aItems, fpProcess, fpFinish) {
 41 		var aCopy = aItems.concat();
 42 		var nIndex = aItems.length - 1;
 43 		var nStart = +new Date();
 44 		setTimeout(function recursive() {
 45 			do {
 46 				nIndex--;
 47 				if (fpProcess(aCopy.shift(), nIndex) === false) {
 48 					return;
 49 				}
 50 			} while (aCopy.length > 0 && (+new Date() - nStart < 50));
 51 
 52 			if (aCopy.length > 0) {
 53 				setTimeout(recursive, 25);
 54 			} else {
 55 				fpFinish(aItems);
 56 			}
 57 		}, 25);
 58 	}
 59 	/**
 60 	 * loop is a function to process items.
 61 	 * @private
 62 	 * @param aItems {Array} Items array to traverse.
 63 	 * @param fpProcess {Function} Callback to execute on each iteration.
 64 	 * @param fpFinish {Function} Callback to execute when all the items are traversed.
 65 	 */
 66 	function loop(aItems, fpProcess, fpFinish)
 67 	{
 68 		var aCopy = aItems.concat();
 69 		var nIndex = aItems.length - 1;
 70 		var oItem = null;
 71 		while(Boolean(oItem = aCopy.shift()))
 72 		{
 73 			nIndex--;
 74 			if(fpProcess(oItem, nIndex) === false)
 75 			{
 76 
 77 				return;
 78 			}
 79 		}
 80 		fpFinish(aItems);
 81 	}
 82 	/**
 83 	 * compare is the function that starts the comparing of image data.
 84 	 * @param aCanvas {Array} Canvas items to execute the compare of image data.
 85 	 * @param fpSuccess {Function} Callback to execute if all the images are equals in pixel level.
 86 	 * @param fpFail {Function} Callback to execute if any of the images is different in pixel level.
 87 	 */
 88 	function compareWithoutCreate(aCanvas, fpSuccess, fpFail, nStart) {
 89 		var sLastData = null,
 90 			nElapsedTime = undefined;
 91 		if(bDebug && typeof nStart === "undefined")
 92 		{
 93 			nStart = +new Date();
 94 		}
 95 		fpLoop(aCanvas, function (oCanvas, nIndex) {
 96 			var oContext = oCanvas.getContext("2d"),
 97 				aCanvasData = oContext.getImageData(0, 0, oCanvas.height, oCanvas.width),
 98 				sData = JSON.stringify([].slice.call(aCanvasData.data));
 99 			if (sLastData !== null) {
100 				if (sLastData.localeCompare(sData) !== 0) {
101 					if(bDebug)
102 					{
103 						oCanvas.className = "fail";
104 						nElapsedTime = (+new Date() - nStart);
105 						console.log("Fail -> Time: " + nElapsedTime);
106 
107 						console.log("Failing  canvas is: ");
108 						console.log(document.getElementById("canvasCompare_" + nIndex));
109 					}
110 					fpFail(oCanvas, nElapsedTime);
111 					return false;
112 				}
113 			}
114 			sLastData = sData;
115 		}, function (aCanvas) {
116 			if(bDebug)
117 			{
118 				nElapsedTime = (+new Date() - nStart);
119 				console.log("Success -> Time: " + nElapsedTime);
120 			}
121 			fpSuccess(aCanvas, nElapsedTime);
122 		});
123 	}
124 	/**
125 	 * createAndCompare creates canvas in oContainer and adding images to these canvas, then compare it
126 	 * @private
127 	 * @param oContainer {Object} Dom element that will contain all the canvas
128 	 * @param aImages {Array} Array of objects that will represent images (
129 	 * @param fpSuccess
130 	 * @param fpFail
131 	 */
132 	function createAndCompare(oContainer, aImages, fpSuccess, fpFail) {
133 		aCanvas = [];
134 		if(bDebug)
135 		{
136 			var nStart = +new Date();
137 		}
138 		fpLoop(aImages, function (oImageConfig, nIndex) {
139 			var oCanvas, oContext, oImage;
140 			oCanvas = document.createElement("canvas");
141 			oCanvas.id = "canvasCompare_" + nIndex;
142 			aCanvas.push(oCanvas);
143 			oCanvas.width = oImageConfig.width;
144 			oCanvas.height = oImageConfig.height;
145 			oContainer.appendChild(oCanvas);
146 			oContext = oCanvas.getContext("2d");
147 			oImage = new Image();
148 			oImage.onload = function()
149 			{
150 				nImagesLoaded++;
151 				oContext.drawImage(oImage, 0, 0);
152 			};
153 			oImage.onerror = function()
154 			{
155 				nImagesLoaded++;
156 			};
157 			oImage.src = oImageConfig.src;
158 		}, function finishCallback(aImages) {
159 			if(nImagesLoaded < aImages.length)
160 			{
161 				setTimeout(function()
162 				{
163 					finishCallback(aImages);
164 				}, 25);
165 			}else
166 			{
167 				compareWithoutCreate(aCanvas, fpSuccess, fpFail, nStart);
168 				if(bDebug && bAsynchronous)
169 				{
170 					console.log("After compare -> Time: " + (+new Date() - nStart));
171 				}
172 			}
173 		});
174 	}
175 	/**
176 	 * ImageToCompare is a JSON helper to create new images objects to be compared.
177 	 * @param sUrl {String} represents the src of the image to be loaded.
178 	 * @param nWidth
179 	 * @param nHeight

181 	var ImageToCompare = function(sUrl, nWidth, nHeight)
182 	{
183 		this.src = sUrl + (sUrl.indexOf("?") === -1? "?": "&") + (+new Date());
184 		this.width = nWidth;
185 		this.height = nHeight;
186 	};
